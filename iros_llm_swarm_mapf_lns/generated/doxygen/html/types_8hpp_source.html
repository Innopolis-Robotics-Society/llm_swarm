47 // Path — sequence of cells indexed by discrete timestep.
48 // path[0] is the position at t=0, path[k] at t=k.

54 // ---------------------------------------------------------------------------
55 // GridMap — static occupancy. blocked[row * cols + col] != 0 means obstacle.
56 // ---------------------------------------------------------------------------
57 struct GridMap {
58  std::size_t rows = 0;
59  std::size_t cols = 0;
60  std::vector<std::uint8_t> blocked; // row-major, 0 = free, 1 = obstacle
61 
62  bool in_bounds(const Cell& c) const {
63  return c.row >= 0 && c.col >= 0 &&
64  static_cast<std::size_t>(c.row) < rows &&
65  static_cast<std::size_t>(c.col) < cols;
66  }
67 
68  bool is_blocked(const Cell& c) const {
69  if (!in_bounds(c)) return true;
70  return blocked[static_cast<std::size_t>(c.row) * cols +
71  static_cast<std::size_t>(c.col)] != 0;
72  }
73 
74  CellIdx index_of(const Cell& c) const {
75  return static_cast<CellIdx>(c.row) * static_cast<CellIdx>(cols) +
76  static_cast<CellIdx>(c.col);
77  }
78 
79  Cell cell_of(CellIdx idx) const {
80  return Cell{static_cast<std::int32_t>(idx / cols),
81  static_cast<std::int32_t>(idx % cols)};
82  }
83 };

86 // FootprintModel — set of cell offsets relative to base_link that the
87 // robot covers when its base cell is c. Includes the (0,0) offset.
88 //
89 // For an arbitrary rotation model footprints could be heading-indexed, but
90 // for v1 we use heading-independent (circular envelope).
91 // ---------------------------------------------------------------------------
92 struct FootprintModel {
93  std::vector<CellOffset> offsets;
94

96  // Rasterizes the disk into a list of offsets.
97  static FootprintModel from_radius(double radius_m, double resolution_m);
98 
99  bool empty() const { return offsets.empty(); }
100  std::size_t size() const { return offsets.size(); }
101 };
