3 // Space-time A* with SOFT collision costs.
4 //
5 // State : (cell, t)
6 // Edge cost : step_cost
7 // + penalty * (# other agents occupying next footprint cells at t+1)
8 // + penalty * (# other agents swap-traversing the edge at t)
9 // h : true distance on the static graph (precomputed backward).
10 // Terminate : when f_min over OPEN >= best known total cost (g + tail).
11 // Returns : path with minimal total cost w.r.t. given CollisionTable and
12 // hold-at-goal tail. Path always ends at goal (unless horizon
13 // exceeded, which is a hard failure).
14 //
15 // NB: penalty must be strictly greater than horizon * step_cost, so the
16 // lex order is (collisions, path_length).

49  public:
50  // Returns h[cell] — the true distance from cell to goal on the static
51  // grid (ignoring agents). Cached by (goal, diag) within a single grid
52  // instance: the cache auto-invalidates if `get()` is called with a
53  // different grid pointer than last time, so a caller mutating the
54  // GridMap's blocked array between solves cannot accidentally reuse a
55  // stale h-table that was built against the old occupancy.
56  //
57  // (The pointer-identity check is conservative — it cannot detect
58  // in-place edits of the same GridMap object. Solvers expect the grid
59  // to be immutable for the duration of a solve, so this is sufficient
60  // in practice; callers who swap blocked cells mid-solve must call
61  // clear() explicitly.)

71  struct Key {
72  std::int32_t row, col;
73  bool diag;
74  bool operator==(const Key& o) const {
75  return row == o.row && col == o.col && diag == o.diag;
76  }
77  };
78  struct KeyHash {
79  std::size_t operator()(const Key& k) const noexcept {
80  std::size_t h = std::hash<std::int64_t>{}(
81  (static_cast<std::int64_t>(k.row) << 32) | (k.col & 0xffffffff));
82  h ^= (k.diag ? 1 : 0);
83  return h;
84  }
85  };
