3 // Bundles the per-robot mission-scoped state previously spread across nine
4 // separate arrays/sets in MapfLns2Node:
5 // * arrival latch
6 // * last-movement baseline (pos + time)
7 // * three "consecutive failure" counters
8 // * lives remaining
9 // * unplanable membership and bench timestamp
10 //
11 // Read access goes through the public fields. The intricate mutation
12 // clusters (fresh-mission reset, bench-on-stall, revive-after-cooldown)
13 // are exposed as methods so the callers don't have to keep the field
14 // invariants straight by hand.
15 //
16 // Thread-safety is the caller's responsibility — every public method must
17 // be called under the same mutex (state_mutex_) that protects the rest of
18 // the mission state.

33 struct RobotLifecycle {
34  using Pos = std::pair<double, double>;
35 
36  // ---- Per-robot arrays (size N == num_robots; indexed by external id) ---
37  std::vector<bool> arrived; // latched once, used for one-shot logs
38  std::vector<Pos> last_move_pos; // last position past the stall threshold
39  std::vector<rclcpp::Time> last_move_time; // time of that last position
40  std::vector<int> empty_fails; // consecutive empty-path solves
41  std::vector<int> stall_iters; // consecutive stalled check_schedule ticks
42  std::vector<int> failed_ticks; // consecutive STATE_FAILED ticks
43  std::vector<int> lives; // unplanable events left before permanent
44 
45  // ---- Cross-cutting sets keyed by external robot_id --------------------
46  std::unordered_set<std::uint32_t> unplanable;
47  std::unordered_map<std::uint32_t, rclcpp::Time> unplanable_since;
48 
49  int max_lives = 0;
50  int num_robots() const { return static_cast<int>(lives.size()); }
51 
52  // ---- Lifecycle clusters -----------------------------------------------
53 
54  // One-time setup; called from the node constructor.
55  void init(int num_robots_arg, int max_lives_arg);
56 
57  // Reset everything that's mission-scoped: counters → 0, lives → max,
58  // arrival latches dropped, unplanable bookkeeping cleared.
59  void reset_for_new_mission();
60 
61  enum class BenchResult { Benched, Permanent, AlreadyUnplanable };
62 
63  // Bench a robot whose stall / empty-fail counter has crossed the
64  // threshold. Decrements `lives[rid]`; if any are left, marks the robot
65  // benched (it will be revived after cooldown), else marks it permanent.
66  // Returns AlreadyUnplanable when the robot was already in the set
67  // (idempotent — no-op).
68  BenchResult bench(std::uint32_t rid, const rclcpp::Time& t);
69 
70  // Restore a benched robot to the planning pool: clear its unplanable
71  // mark, zero its stall / failure counters, refresh movement baseline.
72  // Caller must have already verified rid was benched.
73  void revive(std::uint32_t rid, const Pos& current_pos,
74  const rclcpp::Time& t);
75 
76  // Refresh movement baseline. Called when the robot moves further than
77  // stall_move_thresh_m_ since the last sample, and on arrival.
78  void mark_movement(std::uint32_t rid, const Pos& pos,
79  const rclcpp::Time& t);
80 };
