3 // Footprint-aware vertex + edge occupancy tracker with incremental updates
4 // and O(1) count lookups for the soft A*.
5 //
6 // Invariants:
7 // * total_collisions() is the sum over all unordered agent pairs (i, j),
8 // i != j, of the number of (cell, t) entries where both i and j occupy
9 // the same footprint cell at time t (vertex), plus the number of
10 // (edge, t) swaps where i goes c->c' while j goes c'->c.
11 // * per_agent_collisions[i] is the sum over j of pair collisions (i, j).
12 // Thus sum(per_agent) == 2 * total_collisions().
13 // * touches_[i] is a superset of the set of agents j that currently
14 // share at least one (cell, t) or swap edge with i.
15 // (It may over-approximate after removals; rebuild_touches() resets.)

113  // Number of OTHER agents occupying `cell` at time `t`, excluding `requester`.
114  std::size_t vertex_count_at(CellIdx cell, Timestep t,
115  AgentId requester) const {
116  auto it = vertex_occ_.find(CellTime{cell, t});
117  if (it == vertex_occ_.end()) return 0;
118  std::size_t n = it->second.size();
119  // subtract requester if it is listed (shouldn't be during repair, but
120  // safe to check)
121  for (AgentId a : it->second) if (a == requester) { --n; break; }
122  return n;
123  }

127  std::size_t edge_swap_count(CellIdx from, CellIdx to, Timestep t,
