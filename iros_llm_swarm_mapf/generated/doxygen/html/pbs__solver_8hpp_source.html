32 class ConflictDetector {
33  public:
34  // Detect conflicts using capsule overlap of movement segments.
35  // Each agent sweeps a capsule (circle along a line segment) per timestep.
36  // Capsule overlap catches both positional overlap AND mid-step crossing
37  // conflicts that point-based checks miss with diagonal/multi-cell moves.
38  // Soft-zone proximity is handled by A* penalties, not PBS branching.
39  static std::optional<Conflict> find_first(
40  const std::vector<Path>& paths,
41  const std::vector<float>& footprint_cells) {
42  const size_t n = paths.size();
43  const size_t T = max_len(paths);
44 
45  // Precompute grace periods for overlapping-start pairs.
46  // Agents whose starts overlap (at hard radius) get
47  // ceil(combined_radius - distance) + 1 timesteps to separate
48  // before conflicts are enforced.
49  std::vector<std::vector<size_t>> grace(n, std::vector<size_t>(n, 0));
50  for (size_t i = 0; i < n; ++i)
51  for (size_t j = i + 1; j < n; ++j) {
52  const size_t g = start_grace(paths[i][0], footprint_cells[i],
53  paths[j][0], footprint_cells[j]);
54  grace[i][j] = grace[j][i] = g;
55  }
56 
57  for (size_t t = 0; t < T; ++t) {
58  for (size_t i = 0; i < n; ++i) {
59  for (size_t j = i + 1; j < n; ++j) {
60  if (t < grace[i][j]) continue;
61 
62  // Build movement segments for this timestep
63  const Cell ci_from = at(paths[i], t > 0 ? t - 1 : 0);
64  const Cell ci_to = at(paths[i], t);
65  const Cell cj_from = at(paths[j], t > 0 ? t - 1 : 0);
66  const Cell cj_to = at(paths[j], t);
67 
68  const Segment si = Segment::from_cells(ci_from, ci_to);
69  const Segment sj = Segment::from_cells(cj_from, cj_to);
70 
71  if (capsule_overlap(si, footprint_cells[i], sj, footprint_cells[j])) {
72  return Conflict{i, j, t};
73  }
74  }
75  }
76  }
77  return std::nullopt;
78  }
79 
80  // How many timesteps two overlapping agents need to separate.
81  // Returns 0 if starts don't overlap.
82  static size_t start_grace(const Cell& a, float ra, const Cell& b, float rb) {
83  const float combined = ra + rb;
84  if (combined <= 0.0f) return 0;
85  const float dr = static_cast<float>(a.row) - static_cast<float>(b.row);
86  const float dc = static_cast<float>(a.col) - static_cast<float>(b.col);
87  const float dist = std::sqrt(dr * dr + dc * dc);
88  if (dist >= combined) return 0;
89  return static_cast<size_t>(std::ceil(combined - dist)) + 1;
90  }
91 
92  private:
93  static Cell at(const Path& p, size_t t) {
94  if (p.empty()) return {};
95  return t < p.size() ? p[t] : p.back();
96  }
97  static size_t max_len(const std::vector<Path>& ps) {
98  size_t r = 0;
99  for (const auto& p : ps) r = std::max(r, p.size());
100  return r;
101  }
102 };

108 class PriorityGraph {
109  public:
110  explicit PriorityGraph(size_t n = 0) : n_(n), m_(n * n, 0) {}
111 
112  size_t size() const { return n_; }
113 
114  bool has(size_t hi, size_t lo) const { return at(hi, lo); }
115 
116  bool add(size_t hi, size_t lo) {
117  if (hi == lo) return false;
118  if (has(lo, hi)) return false;
119  if (has(hi, lo)) return true;
120 
121  std::vector<size_t> preds, succs;
122  for (size_t i = 0; i < n_; ++i) if (i == hi || has(i, hi)) preds.push_back(i);
123  for (size_t j = 0; j < n_; ++j) if (j == lo || has(lo, j)) succs.push_back(j);
124  for (size_t p : preds) for (size_t s : succs) if (p == s) return false;
125  for (size_t p : preds) for (size_t s : succs) set(p, s, 1);
126  return true;
127  }
128 
129  std::vector<size_t> higher_than(size_t a) const {
130  std::vector<size_t> r;
131  for (size_t i = 0; i < n_; ++i) if (has(i, a)) r.push_back(i);
132  return r;
133  }
134 
135  std::vector<size_t> topo_from(size_t root) const {
136  std::vector<uint8_t> sel(n_, 0);
137  sel[root] = 1;
138  for (size_t v = 0; v < n_; ++v) if (has(root, v)) sel[v] = 1;
139 
140  std::vector<int> indeg(n_, 0);
141  for (size_t u = 0; u < n_; ++u) {
142  if (!sel[u]) continue;
143  for (size_t v = 0; v < n_; ++v) if (sel[v] && has(u, v)) ++indeg[v];
144  }
145 
146  std::deque<size_t> q;
147  for (size_t v = 0; v < n_; ++v) if (sel[v] && indeg[v] == 0) q.push_back(v);
148 
149  std::vector<size_t> order;
150  while (!q.empty()) {
151  size_t v = q.front(); q.pop_front();
152  order.push_back(v);
153  for (size_t to = 0; to < n_; ++to) {
154  if (sel[to] && has(v, to) && --indeg[to] == 0) q.push_back(to);
155  }
156  }
157  return order;
158  }
159 
160  size_t hash() const {
161  size_t h = 0;
162  for (size_t i = 0; i < m_.size(); i += 8) {
163  uint64_t b = 0;
164  for (size_t j = 0; j < 8 && i+j < m_.size(); ++j)
165  b |= static_cast<uint64_t>(m_[i+j] != 0) << j;
166  h ^= std::hash<uint64_t>{}(b) + 0x9e3779b9 + (h << 6) + (h >> 2);
167  }
168  return h;
169  }
170 
171  bool operator==(const PriorityGraph& o) const { return n_ == o.n_ && m_ == o.m_; }
172 
173  private:
174  size_t n_;
175  std::vector<uint8_t> m_;
176 
177  uint8_t at(size_t i, size_t j) const { return m_[i * n_ + j]; }
178  void set(size_t i, size_t j, uint8_t v) { m_[i * n_ + j] = v; }
179 };

206  // Main entry point. Returns true if a solution is found.
207  // solution[i] — path for agents[i], each element is a Cell in grid coordinates.
208  // resolution — PBS grid cell size in metres (for radius conversion).
209  // stats — if non-null, filled with solve statistics.
210  // cost_curve — gradient shape (Linear, Quadratic, Cubic).
211  // proximity_penalty — cost per step at the hard boundary (0 at soft boundary).
212  // max_astar_expansions — per-A* call expansion limit (0 = unlimited).
213  bool solve(const std::vector<Agent>& agents, std::vector<Path>& solution,
214  float resolution = 0.0f, SolveStats* stats = nullptr,
215  size_t max_expansions = 5000,
216  CostCurve cost_curve = CostCurve::Quadratic,
217  int proximity_penalty = 50,
218  size_t max_astar_expansions = 200000) {
219  solution.clear();
220  if (!map_ || agents.empty()) return false;
221 
222  const size_t n = agents.size();
223  stats_ = stats;
224  cost_curve_ = cost_curve;
225  proximity_penalty_ = proximity_penalty;
226  max_astar_expansions_ = max_astar_expansions;
227 
228  // Precompute radii in cells: footprint (hard) and soft (footprint+inflation)
229  footprint_cells_.resize(n);
230  soft_cells_.resize(n);
231  for (size_t i = 0; i < n; ++i) {
232  footprint_cells_[i] = (resolution > 0.0f && agents[i].footprint_radius > 0.0f)
233  ? agents[i].footprint_radius / resolution
234  : 0.0f;
235  soft_cells_[i] = (resolution > 0.0f)
236  ? (agents[i].footprint_radius + agents[i].inflation) / resolution
237  : 0.0f;
238  }
239 
240  // Ensure footprint radius is large enough for the PBS grid resolution
241  // to prevent wall clipping with diagonal/multi-cell moves.
242  for (size_t i = 0; i < n; ++i) {
243  if (resolution > 0.0f && agents[i].footprint_radius > 0.0f) {
244  assert(agents[i].footprint_radius >= 0.7f * resolution &&
245  "footprint_radius must be >= 0.7 * pbs_resolution to prevent wall clipping");
246  }
247  }
248 
249  // Cache gradient-inflated maps for each unique footprint_radius.
250  // Smaller robots fit through gaps that larger formations cannot.
251  // Hard boundary = footprint (determines blocked cells),
252  // soft boundary = footprint + inflation (gradient penalty zone).
253  for (size_t i = 0; i < n; ++i) {
254  const float hard = agents[i].footprint_radius;
255  const float soft = hard + agents[i].inflation;
256  if (hard > 0.0f && inflated_cache_.find(hard) == inflated_cache_.end()) {
257  inflated_cache_[hard] = map_->inflate_gradient(
258  hard, soft, resolution,
259  proximity_penalty, cost_curve);
260  }
261  }
262 
263  // Precompute Dijkstra distances from each agent's goal on its
264  // gradient map. Used as A* heuristic and for agent ordering.
265  // Cached by (goal_cell, footprint_radius) — agents with different
266  // hard radii have different blocked cells and different distances.
267  agent_goal_dists_.resize(n);
268  for (size_t i = 0; i < n; ++i) {
269  set_agent_map(agents[i]);
270  const size_t goal_idx = agents[i].goal.row * map_->cols + agents[i].goal.col;
271  const size_t key = goal_idx
272  ^ (std::hash<float>{}(agents[i].footprint_radius) * 2654435761u);
273  if (dist_cache_.find(key) == dist_cache_.end()) {
274  dist_cache_[key] = dijkstra_from(*low_level_.current_map(),
275  agents[i].goal, move_set_,
276  time_cost_, resolution_);
277  }
278  agent_goal_dists_[i] = &dist_cache_[key];
279  }
280 
281  // Per-agent true distances (Dijkstra cost from start to goal).
282  // Used for branch ordering and root path ordering.
283  agent_dists_.resize(n);
284  size_t max_dist = 0;
285  for (size_t i = 0; i < n; ++i) {
286  const size_t si = agents[i].start.row * map_->cols + agents[i].start.col;
287  const float d = (*agent_goal_dists_[i])[si];
288  agent_dists_[i] = d < std::numeric_limits<float>::max() / 4.0f
289  ? static_cast<size_t>(d) : 0;
290  max_dist = std::max(max_dist, agent_dists_[i]);
291  }
292 
293  // max_t: time horizon for Space-Time A*.
294  // Based on Chebyshev cell distance (min steps in 8-connected grid),
295  // not cost units. Multiplier accounts for obstacle detours.
296  // TODO: per-agent time cap based on individual cell distance
297  // instead of shared max — short-path agents don't need the
298  // full horizon, reducing their state space.
299  // Critical: a 600x600 map = 360k cells; at max_t=1000
300  // Space-Time A* allocates 360M states and hangs.
301  size_t max_cell_dist = 0;
302  for (size_t i = 0; i < n; ++i) {
303  const size_t dr = agents[i].start.row > agents[i].goal.row
304  ? agents[i].start.row - agents[i].goal.row
305  : agents[i].goal.row - agents[i].start.row;
306  const size_t dc = agents[i].start.col > agents[i].goal.col
307  ? agents[i].start.col - agents[i].goal.col
308  : agents[i].goal.col - agents[i].start.col;
309  max_cell_dist = std::max(max_cell_dist, std::max(dr, dc));
310  }
311  const size_t max_t = std::min<size_t>(max_cell_dist * 2 + 32, 600);
312 
313  // Root node: sequential root planning.
314  // Plan agents by decreasing distance to goal — longest paths get
315  // the most freedom, stationary agents (dist=0) plan last and must
316  // yield to moving agents.
317  PBSNode root;
318  root.pg = PriorityGraph(n);
319  root.paths.resize(n);
320 
321  if (stats) stats->diag.max_t_used = max_t;
322 
323  std::vector<size_t> root_order(n);
324  for (size_t i = 0; i < n; ++i) root_order[i] = i;
325  std::sort(root_order.begin(), root_order.end(), [&](size_t a, size_t b) {
326  return agent_dists_[a] > agent_dists_[b];
327  });
328 
329  SegmentReservationTable root_res;
330  for (size_t idx : root_order) {
331  set_agent_map(agents[idx]);
332  low_level_.set_goal_dists(agent_goal_dists_[idx]);
333  // Root planning: uncapped A* — quality root paths prevent PBS explosion.
334  if (!low_level_.find_path(agents[idx].start, agents[idx].goal,
335  root_res, footprint_cells_[idx], soft_cells_[idx],
336  max_t, root.paths[idx],
337  cost_curve, proximity_penalty)) {
338  record_astar(false);
339  // Fallback: A* without reservations. PBS resolves conflicts.
340  SegmentReservationTable empty;
341  if (!low_level_.find_path(agents[idx].start, agents[idx].goal,
342  empty, footprint_cells_[idx], soft_cells_[idx],
343  max_t, root.paths[idx],
344  cost_curve, proximity_penalty)) {
345  record_astar(false);
346  if (stats) {
347  stats->diag.fail_reason = FailReason::RootPathFailed;
348  stats->diag.fail_agent = idx;
349  }
350  return false;
351  }
352  record_astar(true);
353  } else {
354  record_astar(true);
355  }
356  // Reserve this agent's path so subsequent agents avoid it
357  root_res.reserve_path(root.paths[idx], max_t,
358  footprint_cells_[idx], soft_cells_[idx]);
359  }
360  root.cost = cost(root.paths);
361 
362  // PBS best-first search: expand lowest-cost node, detect first
363  // conflict, branch into two children (one per priority ordering).
364  // Visited set prevents re-exploring identical priority graphs.
365  std::priority_queue<PBSNode, std::vector<PBSNode>, PBSNodeCmp> open;
366  std::unordered_set<NodeKey, NodeKeyHash> visited;
367 
368  open.push(root);
369  visited.insert(make_key(root));
370 
371  size_t expansions = 0;
372 
373  while (!open.empty()) {
374  if (++expansions > max_expansions) {
375  if (stats) {
376  stats->expansions = expansions;
377  stats->diag.fail_reason = FailReason::MaxExpansions;
378  }
379  return false;
380  }
381 
382  PBSNode node = open.top(); open.pop();
383 
384  const auto c = ConflictDetector::find_first(
385  node.paths, footprint_cells_);
386  if (!c) {
387  solution = node.paths;
388  if (stats) {
389  stats->expansions = expansions;
390  stats->max_path_length = 0;
391  for (const auto& p : solution)
392  stats->max_path_length = std::max(stats->max_path_length, p.size());
393  }
394  return true;
395  }
396 
397  // TODO: PBS conflict resolution is brute force — just tries both
398  // priority orderings. Needs smarter strategy (CBS, ECBS, or
399  // improved heuristics) to scale beyond simple scenarios.
400  // Branch ordering heuristic: try giving priority to the agent
401  // with the further goal first — it needs more space/freedom.
402  size_t hi = c->agent1, lo = c->agent2;
403  if (agent_dists_[lo] > agent_dists_[hi]) std::swap(hi, lo);
404 
405  // branch 1: further-goal agent gets priority
406  PBSNode ch1 = node;
407  if (branch(agents, hi, lo, max_t, ch1)) {
408  auto k = make_key(ch1);
409  if (visited.insert(k).second) { ch1.cost = cost(ch1.paths); open.push(std::move(ch1)); }
410  if (stats) ++stats->diag.branches_tried;
411  } else {
412  if (stats) ++stats->diag.branches_failed;
413  }
414 
415  // branch 2: closer-goal agent gets priority
416  PBSNode ch2 = std::move(node);
417  if (branch(agents, lo, hi, max_t, ch2)) {
418  auto k = make_key(ch2);
419  if (visited.insert(k).second) { ch2.cost = cost(ch2.paths); open.push(std::move(ch2)); }
420  if (stats) ++stats->diag.branches_tried;
421  } else {
422  if (stats) ++stats->diag.branches_failed;
423  }
424  }
425  if (stats) {
426  stats->expansions = expansions;
427  stats->diag.fail_reason = FailReason::BranchExhausted;
428  }
429  return false;

521  size_t max_t, PBSNode& node) {

523  for (size_t ai : order) {

525  // For agents whose starts overlap with ours, skip reservation

528  for (size_t bi : node.pg.higher_than(ai)) {
529  const size_t grace = ConflictDetector::start_grace(
530  agents[ai].start, footprint_cells_[ai],
531  agents[bi].start, footprint_cells_[bi]);
532  res.reserve_path(node.paths[bi], max_t,
533  footprint_cells_[bi], soft_cells_[bi], grace);
534  }

539  Path p;
540  if (!low_level_.find_path(agents[ai].start, agents[ai].goal,
541  res, footprint_cells_[ai], soft_cells_[ai],
542  max_t, p,

544  max_astar_expansions_)) {
545  record_astar(false);
546  return false;
547  }
548  record_astar(true);
