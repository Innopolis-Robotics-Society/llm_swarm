13 // Stores centres and radii of higher-priority agents.
14 // Conflict check via Euclidean distance between centres:
15 // if distance < (r_a + r_b), agents are considered in conflict.

25  // Reserve agent path with the given radius (in cells).
26  // footprint_cells: physical (hard) radius in cells.
27  // soft_radius_cells: footprint + inflation in cells (outer penalty boundary).
28  // skip_until: don't reserve entries for t < skip_until (grace period
29  // for agents whose starts overlap — lets them separate first).
30  void reserve_path(const Path& path,
31  size_t hold_goal_until_time,
32  float footprint_cells, float soft_radius_cells,
33  size_t skip_until = 0) {
34  if (path.empty()) return;
35  for (size_t t = skip_until; t < path.size(); ++t) {
36  entries_[t].push_back({path[t], footprint_cells, soft_radius_cells});
37  }
38  // Goal hold: agent stays in place after arrival
39  if (path.size() <= hold_goal_until_time) {
40  held_goals_.push_back({path.back(), footprint_cells, soft_radius_cells,
41  std::max(path.size(), skip_until)});
42  }
43  }
44 
45  // Gradient penalty for placing an agent at (row,col) at time t.
46  // Returns: 0 = free, >0 = soft penalty, -1 = FORBIDDEN (physical overlap).
47  // my_footprint/my_soft: agent's own radii in cells.
48  // max_penalty: cost at the hard boundary (decreasing to 0 at soft boundary).
49  // Multiple nearby agents: takes the greatest penalty, not sum.
50  int vertex_penalty(size_t row, size_t col, size_t time,
51  float my_footprint, float my_soft,
52  int max_penalty = 50,
53  CostCurve cost_curve = CostCurve::Quadratic) const {
54  int total = 0;
55  auto check = [&](const Cell& c, float fp, float sr) {
56  const float dr = static_cast<float>(row) - static_cast<float>(c.row);
57  const float dc = static_cast<float>(col) - static_cast<float>(c.col);
58  const float dist_sq = dr * dr + dc * dc;
59  const float hard = my_footprint + fp;
60  const float soft = my_soft + sr;
61  if (hard > 0.0f && dist_sq < hard * hard) { total = -1; return; }
62  if (soft > 0.0f && dist_sq < soft * soft) {
63  const float dist = std::sqrt(dist_sq);
64  const float ratio = (soft - dist) / (soft - hard);
65  const int cost = static_cast<int>(
66  apply_curve(ratio, cost_curve) * max_penalty);
67  total = std::max(total, cost);
68  }
69  };
70 
71  // Check held goals
72  for (const auto& hg : held_goals_) {
73  if (time >= hg.from_time) {
74  check(hg.cell, hg.footprint_cells, hg.soft_radius_cells);
75  if (total < 0) return -1;
76  }
77  }
78  // Check positions at the specific timestep
79  auto it = entries_.find(time);
80  if (it != entries_.end()) {
81  for (const auto& e : it->second) {
82  check(e.cell, e.footprint_cells, e.soft_radius_cells);
83  if (total < 0) return -1;
84  }
85  }
86  return total;
87  }
88 
89  // Edge conflict: checks opposing movement between steps time and time+1.
90  // Uses footprint (hard) radius only — physical swap prevention.
91  bool is_edge_blocked(size_t fr, size_t fc, size_t tr, size_t tc,
92  size_t time, float my_footprint_cells) const {
93  auto it = entries_.find(time);
94  auto it_next = entries_.find(time + 1);
95  if (it == entries_.end() || it_next == entries_.end()) return false;
96 
97  // For each agent: check if it moves in the opposite direction.
98  // NOTE: pairing by vector index assumes reserve_path() is called
99  // once per agent in a fixed order. With skip_until, an agent's
100  // entries start later, so indices at the grace boundary may differ —
101  // the min(size, size) guard safely skips unmatched trailing entries.
102  // TODO: edge conflicts are silently missed at the grace boundary
103  // timestep. Fix by pairing entries by agent ID, not vector index.
104  for (size_t i = 0; i < it->second.size() && i < it_next->second.size(); ++i) {
105  const auto& cur = it->second[i];
106  const auto& next = it_next->second[i];
107  // Edge conflict: after swap both agents end up too close
108  // to each other's previous position
109  if (cells_conflict(fr, fc, my_footprint_cells,
110  next.cell.row, next.cell.col, cur.footprint_cells) &&
111  cells_conflict(tr, tc, my_footprint_cells,
112  cur.cell.row, cur.cell.col, cur.footprint_cells))
113  return true;
114  }
115  return false;
116  }
117 
118  // Can the agent hold its goal starting from from_time?
119  // Uses footprint (hard) radius — agent can hold goal as long as no
120  // physical overlap with other agents' paths or held goals.
121  bool can_hold_goal(size_t row, size_t col, size_t from_time,
122  size_t max_time, float my_footprint_cells) const {
123  for (const auto& hg : held_goals_) {
124  if (cells_conflict(row, col, my_footprint_cells,
125  hg.cell.row, hg.cell.col, hg.footprint_cells))
126  return false;
127  }
128  for (size_t t = from_time; t <= max_time; ++t) {
129  auto it = entries_.find(t);
130  if (it == entries_.end()) continue;
131  for (const auto& e : it->second) {
132  if (cells_conflict(row, col, my_footprint_cells,
133  e.cell.row, e.cell.col, e.footprint_cells))
134  return false;
135  }
136  }
137  return true;
138  }

156  // Two-circle conflict check: distance between centres < sum of radii.
157  // With zero radii — point-based cell coincidence check.
158  static bool cells_conflict(size_t r1, size_t c1, float rad1,
159  size_t r2, size_t c2, float rad2) {
160  const float min_dist = rad1 + rad2;
161  if (min_dist <= 0.0f) {
162  return r1 == r2 && c1 == c2;
163  }
164  const float dr = static_cast<float>(r1) - static_cast<float>(r2);
165  const float dc = static_cast<float>(c1) - static_cast<float>(c2);
166  return dr * dr + dc * dc < min_dist * min_dist;
167  }

202  if (is_blocked(start.row, start.col) || is_blocked(goal.row, goal.col)) return false;
203

205  const size_t state_count = spatial * (max_time + 1);
206

208  next_generation();
209 
210  const size_t start_idx = idx(start.row, start.col);
211  const size_t goal_idx = idx(goal.row, goal.col);
212  const size_t start_state = state_idx(start_idx, 0, spatial);
213 
214  // NOTE: we intentionally do NOT reject starts that overlap with
215  // reservations at t=0. During replanning agents are physically at
216  // their current positions and must be allowed to start there even if
217  // another agent's reservation covers the same cell. A* will
218  // naturally find a path that diverges at t=1+.
219 
220  set_g(start_state, 0);
221  parent_[start_state] = INVALID;
222 
223  std::priority_queue<Node, std::vector<Node>, NodeCmp> open;
224  open.push({start_idx, 0, 0, heuristic(start, goal)});
225 
226  last_expansions_ = 0;
227  while (!open.empty()) {
228  ++last_expansions_;
229  if (max_astar_expansions > 0 && last_expansions_ > max_astar_expansions)
230  return false;
231 
232  const Node cur = open.top(); open.pop();
233  const size_t cs = state_idx(cur.idx, cur.t, spatial);
234 
235  if (is_closed(cs)) continue;
236  if (!is_alive(cs) || cur.g != g_[cs]) continue;
237  mark_closed(cs);
238 
239  if (cur.idx == goal_idx) {
240  if (reservations.can_hold_goal(goal.row, goal.col, cur.t,
241  max_time, my_footprint_cells)) {
242  restore(cs, spatial, path);
243  return true;
244  }
245  }
246 
247  if (cur.t >= max_time) continue;
248 
249  const Cell c = cell(cur.idx);
250  const size_t r = c.row, col_ = c.col;
251  const size_t nt = cur.t + 1;
252 
253  // wait in place
254  relax(cur.idx, cur.t, r, col_, nt, goal, reservations, open, spatial);
255  // neighbours
256  if (r > 0) relax(cur.idx, cur.t, r-1, col_, nt, goal, reservations, open, spatial);
257  if (r+1 < map_->rows) relax(cur.idx, cur.t, r+1, col_, nt, goal, reservations, open, spatial);
258  if (col_ > 0) relax(cur.idx, cur.t, r, col_-1, nt, goal, reservations, open, spatial);
259  if (col_+1 < map_->cols) relax(cur.idx, cur.t, r, col_+1, nt, goal, reservations, open, spatial);
260  }

262  }

300  // Heuristic for A*: precomputed Dijkstra step-distance from goal.
301  // Accounts for wall topology (blocked cells) but excludes penalties.
302  // Admissible — always <= true cost. Falls back to Manhattan if
303  // goal_dists_ is not set (e.g. during testing).
304  int heuristic(const Cell& a, const Cell& b) const {
305  if (goal_dists_) {
306  const int d = (*goal_dists_)[idx(a.row, a.col)];
307  return d < INF ? d : INF;
308  }
309  const int dr = a.row > b.row ? a.row - b.row : b.row - a.row;
310  const int dc = a.col > b.col ? a.col - b.col : b.col - a.col;
311  return dr + dc;
312  }

330  void relax(size_t ci, size_t ct, size_t nr, size_t nc, size_t nt,
331  const Cell& goal, const ReservationTable& res,
332  std::priority_queue<Node, std::vector<Node>, NodeCmp>& open,
333  size_t sp) {
334  if (is_blocked(nr, nc)) return;
335 
336  const size_t ni = idx(nr, nc);
337  const bool moving = (ni != ci);
338 
339  // Wall penalty only when entering a new cell (not waiting).
340  // Agent penalty every timestep (occupying space near them).
341  int penalty = moving ? wall_penalty(nr, nc) : 0;
342 
343  const int agent_pen = res.vertex_penalty(
344  nr, nc, nt, my_footprint_, my_soft_,
345  proximity_penalty_, cost_curve_);
346  if (agent_pen < 0) return; // physical agent overlap forbidden
347  penalty += agent_pen;
348 
349  // Edge conflict: check opposing movement (uses footprint/hard radius)
350  const Cell from_cell = cell(ci);
351  if (res.is_edge_blocked(from_cell.row, from_cell.col, nr, nc, ct, my_footprint_))
352  return;
353 
354  const size_t cs = state_idx(ci, ct, sp);
355  const size_t ns = state_idx(ni, nt, sp);
356  if (is_closed(ns)) return;
357 
358  // Every timestep costs 1. Movement adds 1 more — agents prefer
359  // waiting over unnecessary walking.
360  const int tg = get_g(cs) + 1 + (moving ? 1 : 0) + penalty;
361  if (tg < get_g(ns)) {
362  set_g(ns, tg);
363  parent_[ns] = cs;
364  open.push({ni, nt, tg, tg + heuristic({nr, nc}, goal)});
365  }
366  }
