40 struct GridMap {
41  size_t rows = 0;
42  size_t cols = 0;
43  std::vector<uint8_t> blocked; // 1 = hard blocked (wall or within footprint radius)
44  std::vector<int> wall_cost; // gradient penalty near walls (0 = free, >0 = penalty)
45 
46  // Gradient inflation: cells within hard_r are blocked, cells between
47  // hard_r and soft_r get a gradient penalty (max_penalty at hard boundary,
48  // 0 at soft boundary). cost_curve controls the gradient shape.
49  GridMap inflate_gradient(float hard_m, float soft_m, float resolution_m,
50  int max_penalty = 10,
51  CostCurve cost_curve = CostCurve::Quadratic) const {
52  if (soft_m <= 0.0f) return *this;
53  const float hard_r = hard_m / resolution_m;
54  const float soft_r = soft_m / resolution_m;
55  const int scan_r = static_cast<int>(std::ceil(soft_r));
56 
57  GridMap out;
58  out.rows = rows;
59  out.cols = cols;
60  out.blocked.assign(rows * cols, 0);
61  out.wall_cost.assign(rows * cols, 0);
62 
63  const float hard_r_sq = hard_r * hard_r;
64  const float soft_r_sq = soft_r * soft_r;
65 
66  for (size_t row = 0; row < rows; ++row) {
67  for (size_t col = 0; col < cols; ++col) {
68  if (blocked[row * cols + col] == 0) continue;
69  for (int dr = -scan_r; dr <= scan_r; ++dr) {
70  for (int dc = -scan_r; dc <= scan_r; ++dc) {
71  const float dist_sq = static_cast<float>(dr * dr + dc * dc);
72  if (dist_sq > soft_r_sq) continue;
73  const int nr = static_cast<int>(row) + dr;
74  const int nc = static_cast<int>(col) + dc;
75  if (nr < 0 || nr >= static_cast<int>(rows)) continue;
76  if (nc < 0 || nc >= static_cast<int>(cols)) continue;
77  const size_t idx = nr * cols + nc;
78  if (dist_sq <= hard_r_sq) {
79  out.blocked[idx] = 1;
80  } else {
81  const float dist = std::sqrt(dist_sq);
82  const float ratio = (soft_r - dist) / (soft_r - hard_r);
83  const int cost = static_cast<int>(
84  apply_curve(ratio, cost_curve) * max_penalty);
85  out.wall_cost[idx] = std::max(out.wall_cost[idx], cost);
86  }
87  }
88  }
89  }
90  }
91  return out;
92  }
93 };

148 // Minimum squared distance between two line segments.
149 // Each segment is parameterized as P(s) = A + s*(B-A), s in [0,1].
150 inline float segment_segment_dist_sq(const Segment& s1, const Segment& s2) {
151  const float d1r = s1.r1 - s1.r0, d1c = s1.c1 - s1.c0;
152  const float d2r = s2.r1 - s2.r0, d2c = s2.c1 - s2.c0;
153  const float rr = s1.r0 - s2.r0, rc = s1.c0 - s2.c0;
154 
155  const float a = d1r * d1r + d1c * d1c; // |d1|^2
156  const float e = d2r * d2r + d2c * d2c; // |d2|^2
157  const float b = d1r * d2r + d1c * d2c; // d1 . d2
158  const float c = d1r * rr + d1c * rc; // d1 . r
159  const float f = d2r * rr + d2c * rc; // d2 . r
160 
161  // Solve for unconstrained closest points (s, t) on the two lines.
162  // denom = 0 means segments are parallel or degenerate (zero-length).
163  const float denom = a * e - b * b;
164  float s, t;
165 
166  if (denom < 1e-8f) {
167  // Parallel/degenerate: fix s=0, project onto segment 2
168  s = 0.0f;
169  t = (e > 1e-8f) ? f / e : 0.0f;
170  } else {
171  s = (b * f - c * e) / denom;
172  t = (a * f - b * c) / denom;
173  }
174 
175  // Two-pass clamp: clamp s to [0,1], re-derive optimal t,
176  // then clamp t to [0,1], re-derive optimal s. This finds the
177  // true closest point pair when the unconstrained solution falls
178  // outside the segment bounds.
179  s = std::max(0.0f, std::min(1.0f, s));
180  t = (e > 1e-8f) ? (b * s + f) / e : 0.0f;
181 
182  t = std::max(0.0f, std::min(1.0f, t));
183  if (a > 1e-8f) {
184  s = (b * t - c) / a;
185  s = std::max(0.0f, std::min(1.0f, s));
186  }
187 
188  // Squared distance between the closest points on the two segments
189  const float pr = s1.r0 + s * d1r - (s2.r0 + t * d2r);
190  const float pc = s1.c0 + s * d1c - (s2.c0 + t * d2c);
191  return pr * pr + pc * pc;
192 }

194 // Check if two capsules overlap (swept circles along segments).
195 inline bool capsule_overlap(const Segment& a, float radius_a,
196  const Segment& b, float radius_b) {
197  const float combined = radius_a + radius_b;
198  if (combined <= 0.0f) {
199  return segment_segment_dist_sq(a, b) < 1e-6f;
200  }
201  return segment_segment_dist_sq(a, b) < combined * combined;
202 }

204 // Gradient penalty between two capsules.
205 // Returns: -1 = hard overlap (forbidden), 0 = free, >0 = soft penalty.
206 inline int capsule_penalty(const Segment& a, float hard_a, float soft_a,
207  const Segment& b, float hard_b, float soft_b,
208  int max_penalty = 50,
209  CostCurve cost_curve = CostCurve::Quadratic) {
210  const float dist_sq = segment_segment_dist_sq(a, b);
211  const float hard_dist = hard_a + hard_b;
212  const float soft_dist = soft_a + soft_b;
213 
214  if (hard_dist > 0.0f && dist_sq < hard_dist * hard_dist) return -1;
215  if (soft_dist <= 0.0f || dist_sq >= soft_dist * soft_dist) return 0;
216 
217  const float dist = std::sqrt(dist_sq);
218  const float ratio = (soft_dist - dist) / (soft_dist - hard_dist);
219  return static_cast<int>(apply_curve(ratio, cost_curve) * max_penalty);
220 }

235  // Generate all reachable moves from physical parameters.
236  // reach = max(1, floor(max_speed * time_step / resolution))
237  // Includes wait (0,0) and all (dr,dc) where dr²+dc² <= reach².
238  static MoveSet generate(float max_speed, float time_step_sec,
239  float resolution) {
240  const int reach = std::max(1,
241  static_cast<int>(std::floor(max_speed * time_step_sec / resolution)));
242  MoveSet ms;
243  ms.moves.push_back({0, 0, 0.0f}); // wait
244  for (int dr = -reach; dr <= reach; ++dr) {
245  for (int dc = -reach; dc <= reach; ++dc) {
246  if (dr == 0 && dc == 0) continue;
247  if (dr * dr + dc * dc <= reach * reach) {
248  ms.moves.push_back({dr, dc,
249  std::sqrt(static_cast<float>(dr * dr + dc * dc))});
250  }
251  }
252  }
253  return ms;
254  }

261 // Trace cells along line from (r0,c0) to (r1,c1) using Bresenham.
262 // Callback receives each cell (row, col). Returns false from callback to stop early.
263 template <typename Fn>
264 void bresenham_trace(int r0, int c0, int r1, int c1, Fn&& fn) {
265  int dr = r1 > r0 ? r1 - r0 : r0 - r1;
266  int dc = c1 > c0 ? c1 - c0 : c0 - c1;
267  int sr = r0 < r1 ? 1 : -1;
268  int sc = c0 < c1 ? 1 : -1;
269  int err = dr - dc;
270 
271  while (true) {
272  if (!fn(r0, c0)) return;
273  if (r0 == r1 && c0 == c1) break;
274  const int e2 = 2 * err;
275  if (e2 > -dc) { err -= dc; r0 += sr; }
276  if (e2 < dr) { err += dr; c0 += sc; }
277  }
278 }

280 // Check if a move is valid (no blocked cells along trace) and compute
281 // the sum of wall_cost along the trace.
282 // Returns: {valid, wall_penalty_sum}
283 inline std::pair<bool, float> trace_move(const GridMap& map,
284  size_t from_r, size_t from_c,
285  int dr, int dc) {
286  const int tr = static_cast<int>(from_r) + dr;
287  const int tc = static_cast<int>(from_c) + dc;
288 
289  // Bounds check
290  if (tr < 0 || tr >= static_cast<int>(map.rows)) return {false, 0.0f};
291  if (tc < 0 || tc >= static_cast<int>(map.cols)) return {false, 0.0f};
292 
293  // Wait move: no trace, no cost.
294  if (dr == 0 && dc == 0) return {true, 0.0f};
295 
296  bool valid = true;
297  float penalty = 0.0f;
298  bool first = true;
299 
300  bresenham_trace(static_cast<int>(from_r), static_cast<int>(from_c),
301  tr, tc, [&](int r, int c) -> bool {
302  // Skip source cell — penalty was already charged on entry.
303  if (first) { first = false; return true; }
304  const size_t idx = static_cast<size_t>(r) * map.cols + static_cast<size_t>(c);
305  if (map.blocked[idx]) { valid = false; return false; }
306  if (!map.wall_cost.empty()) penalty += static_cast<float>(map.wall_cost[idx]);
307  return true;
308  });
309 
310  return {valid, penalty};
311 }

324  // Reserve an agent's path as a sequence of segments.
325  // footprint_cells: physical (hard) radius in cells.
326  // soft_radius_cells: footprint + inflation in cells.
327  // skip_until: grace period — skip entries for t < skip_until.
328  void reserve_path(const Path& path, size_t hold_goal_until_time,
329  float footprint_cells, float soft_radius_cells,
330  size_t skip_until = 0) {
331  if (path.empty()) return;
332  for (size_t t = std::max(skip_until, size_t(1)); t < path.size(); ++t) {
333  entries_[t].push_back(
334  {path[t - 1], path[t], footprint_cells, soft_radius_cells});
335  }
336  // Also store wait at t=0 (or skip_until) as a point segment
337  if (skip_until == 0) {
338  entries_[0].push_back(
339  {path[0], path[0], footprint_cells, soft_radius_cells});
340  }
341  // Hold goal: agent stays at final cell after arrival
342  if (path.size() <= hold_goal_until_time) {
343  held_goals_.push_back({path.back(), footprint_cells, soft_radius_cells,
344  std::max(path.size(), skip_until)});
345  }
346  }

348  // Check a candidate move segment against reservations.
349  // Returns: 0 = free, >0 = soft penalty, -1 = forbidden.
350  int segment_penalty(const Cell& from, const Cell& to, size_t time,
351  float my_footprint, float my_soft,
352  int max_penalty = 50,
353  CostCurve cost_curve = CostCurve::Quadratic) const {
354  const Segment my_seg = Segment::from_cells(from, to);
355  int worst = 0;
356 
357  auto check = [&](const SegEntry& e) {
358  const Segment other = Segment::from_cells(e.from, e.to);
359  const int pen = capsule_penalty(
360  my_seg, my_footprint, my_soft,
361  other, e.footprint_cells, e.soft_radius_cells,
362  max_penalty, cost_curve);
363  if (pen < 0) { worst = -1; return; }
364  worst = std::max(worst, pen);
365  };

371  const int pen = capsule_penalty(

374  max_penalty, cost_curve);

376  worst = std::max(worst, pen);
377  }
378  }

381  auto it = entries_.find(time);
382  if (it != entries_.end()) {
383  for (const auto& e : it->second) {
384  check(e);
385  if (worst < 0) return -1;
386  }
387  }

389  }
