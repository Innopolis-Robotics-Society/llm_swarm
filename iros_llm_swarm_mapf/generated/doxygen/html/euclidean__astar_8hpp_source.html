10 // Euclidean Space-Time A* with N-connected movement
11 //
12 // Replaces the 4-connected SpaceTimeAStarPlanner with a connectivity-
13 // agnostic planner. The move set is auto-generated from physical
14 // parameters (speed, timestep, resolution) via MoveSet::generate().
15 //
16 // Key differences from the classic planner:
17 // - Float g-values (Euclidean move costs instead of integer step costs)
18 // - Dijkstra heuristic (replaces BFS — correct for non-uniform costs)
19 // - Move loop over MoveSet instead of hardcoded cardinal neighbors
20 // - Bresenham trace validates multi-cell jumps (blocked check + wall penalty sum)
21 // - Segment-based reservation checks via SegmentReservationTable
22 // (replaces separate vertex_penalty + is_edge_blocked)

180  void relax(size_t ci, size_t ct, const Cell& from, const Move& mv,
181  size_t nt, const Cell& goal,
182  const SegmentReservationTable& res,
183  std::priority_queue<Node, std::vector<Node>, NodeCmp>& open,
184  size_t sp) {
185  // Compute target cell
186  const int nr = static_cast<int>(from.row) + mv.dr;
187  const int nc = static_cast<int>(from.col) + mv.dc;
188  if (nr < 0 || nr >= static_cast<int>(map_->rows)) return;
189  if (nc < 0 || nc >= static_cast<int>(map_->cols)) return;
190 
191  const size_t ni = idx(static_cast<size_t>(nr), static_cast<size_t>(nc));
192  const bool moving = (ni != ci);
193  const Cell to_cell = {static_cast<size_t>(nr), static_cast<size_t>(nc)};
194 
195  // Trace the move: check blocked cells and sum wall penalties
196  float wall_pen = 0.0f;
197  if (moving) {
198  auto [valid, wpen] = trace_move(*map_, from.row, from.col, mv.dr, mv.dc);
199  if (!valid) return;
200  wall_pen = wpen;
201  }
202 
203  // Segment-based reservation check (replaces vertex_penalty + is_edge_blocked)
204  const int agent_pen = res.segment_penalty(
205  from, to_cell, nt, my_footprint_, my_soft_,
206  proximity_penalty_, cost_curve_);
207  if (agent_pen < 0) return;
208 
209  const size_t cs = state_idx(ci, ct, sp);
210  const size_t ns = state_idx(ni, nt, sp);
211  if (is_closed(ns)) return;
212 
213  // Cost = time (opportunity cost) + distance (metres) + wall gradient
214  // (sum along trace, normalized to metres) + agent proximity
215  const float tg = get_g(cs) + time_cost_ + resolution_ * mv.cost
216  + wall_pen * resolution_ + static_cast<float>(agent_pen);
217  if (tg < get_g(ns)) {
218  set_g(ns, tg);
219  parent_[ns] = cs;
220  open.push({ni, nt, tg, tg + heuristic(to_cell, goal)});
221  }
222  }

233 //
234 // Backward Dijkstra from goal using the same MoveSet as A*.
235 // Produces correct minimum-cost distances for any connectivity.
236 // With 4-connected integer costs, degenerates to BFS.
237 // One-time cost per unique goal on the downsampled PBS grid.

240 inline std::vector<float> dijkstra_from(const GridMap& map, const Cell& goal,
241  const MoveSet& moves,
242  float time_cost, float resolution) {
243  static constexpr float INF_F = std::numeric_limits<float>::max() / 4.0f;
244  const size_t N = map.rows * map.cols;
245  std::vector<float> dist(N, INF_F);
246 
247  struct DNode { size_t idx; float d; };
248  struct DCmp { bool operator()(const DNode& a, const DNode& b) { return a.d > b.d; } };
249  std::priority_queue<DNode, std::vector<DNode>, DCmp> pq;
250 
251  const size_t gi = goal.row * map.cols + goal.col;
252  dist[gi] = 0.0f;
253  pq.push({gi, 0.0f});
254 
255  while (!pq.empty()) {
256  const auto [ci, d] = pq.top(); pq.pop();
257  if (d > dist[ci]) continue;
258 
259  const size_t r = ci / map.cols;
260  const size_t c = ci % map.cols;
261 
262  for (const auto& mv : moves.moves) {
263  if (mv.dr == 0 && mv.dc == 0) continue; // skip wait
264  // Reverse direction: expanding backwards from goal
265  const int nr = static_cast<int>(r) - mv.dr;
266  const int nc = static_cast<int>(c) - mv.dc;
267  if (nr < 0 || nr >= static_cast<int>(map.rows)) continue;
268  if (nc < 0 || nc >= static_cast<int>(map.cols)) continue;
269 
270  // Validate the reversed move doesn't cross blocked cells
271  auto [valid, _wpen] = trace_move(map, static_cast<size_t>(nr),
272  static_cast<size_t>(nc), mv.dr, mv.dc);
273  if (!valid) continue;
274 
275  const size_t ni = static_cast<size_t>(nr) * map.cols + static_cast<size_t>(nc);
276  const float nd = d + time_cost + resolution * mv.cost;
277  if (nd < dist[ni]) {
278  dist[ni] = nd;
279  pq.push({ni, nd});
280  }
281  }
282  }
283  return dist;
284 }
